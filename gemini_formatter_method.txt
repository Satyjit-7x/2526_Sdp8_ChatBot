def format_sql_response_with_gemini(self, user_query, sql_type, sql_result, sql_query=""):
    """Format SQL operation results into user-friendly messages using Gemini"""
    try:
        # Get context from ChromaDB
        rag_response = ""
        try:
            results = self.collection.query(query_texts=[user_query], n_results=2)
            if results and results.get('documents') and len(results['documents']) > 0:
                rag_response = "\\n".join(results['documents'][0])
        except:
            pass
        
        # Build context for Gemini
        if sql_type == "CREATE":
            # Extract order details from SQL
            import re
            order_id_match = re.search(r"'(ORD\\d+)'", sql_query)
            product_match = re.search(r"product_name.*?'([^']+)'", sql_query)
            price_match = re.search(r"price.*?(\\d+(?:\\.\\d+)?)", sql_query)
            
            order_id = order_id_match.group(1) if order_id_match else "Unknown"
            product = product_match.group(1) if product_match else "Unknown"
            price = float(price_match.group(1)) if price_match else 0
            
            prompt = (
                f'User requested: "{user_query}"\\n\\n'
                f'Action taken: Created a new order\\n'
                f'- Order ID: {order_id}\\n'
                f'- Product: {product}\\n'
                f'- Price: ₹{price:,.0f}\\n'
                f'- Status: Pending\\n\\n'
                f'Additional context from knowledge base:\\n{rag_response}\\n\\n'
                'Create a friendly, concise response (2-3 sentences max) that:\\n'
                '1. Confirms the order was created successfully\\n'
                '2. Mentions the order ID, product name, and price in ₹\\n'
                '3. Includes helpful next steps or relevant info from knowledge base\\n'
                '4. Sounds natural and conversational\\n\\n'
                'Be brief and friendly!'
            )

        elif sql_type == "UPDATE":
            prompt = (
                f'User requested: "{user_query}"\\n\\n'
                f'Action taken: Updated an order status\\n\\n'
                f'Context: {rag_response}\\n\\n'
                'Create a brief, friendly confirmation message (1-2 sentences) that the order was updated successfully. Include any helpful context.'
            )

        elif sql_type == "DELETE":
            prompt = (
                f'User requested: "{user_query}"\\n\\n'
                f'Action taken: Deleted an order\\n\\n'
                f'Context: {rag_response}\\n\\n'
                'Create a brief, friendly confirmation message (1-2 sentences) that the order was removed. Sound empathetic and helpful.'
            )
        
        else:
            return "Done! Let me know if you need anything else."
        
        # Get Gemini response
        response = self.model.generate_content(prompt)
        return response.text.strip()
        
    except Exception as e:
        print(f"Gemini formatting error: {e}")
        # Fallback to simple message
        if sql_type == "CREATE":
            return "✅ Order created successfully! Your order has been placed."
        elif sql_type == "UPDATE":
            return "✅ Order updated successfully!"
        elif sql_type == "DELETE":
            return "✅ Order deleted successfully!"
        else:
            return "Done!"
